# Contrast Fixes + Light Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Fix subthreshold contrast and undefined-CSS-var bugs in the desktop app, and add a user-toggleable light theme persisted in DB.

**Architecture:** Two-theme token system on `<html data-theme>`. Dark values live on `:root` (also serves as first-paint fallback); `[data-theme="light"]` overrides. SQLite-persisted preference (`'light' | 'dark' | 'system'`) read at boot via tRPC, applied to `documentElement.dataset.theme`. xterm and Monaco swap reactively.

**Tech Stack:** Tailwind v4 (arbitrary `bg-[var(--*)]`), CSS custom props, Drizzle/SQLite, tRPC over Electron IPC, Zustand, monaco-editor, xterm.js.

**Spec:** `docs/superpowers/specs/2026-04-25-contrast-and-light-mode-design.md`

Create `apps/desktop/src/main/trpc/routers/settings.ts`:

```ts
import { eq } from "drizzle-orm";
import { BrowserWindow } from "electron";
import { z } from "zod";
import { db } from "../../db";
import { appSettings } from "../../db/schema";
import { publicProcedure, router } from "../index";

const themeSchema = z.enum(["system", "light", "dark"]);
export type ThemePref = z.infer<typeof themeSchema>;

const THEME_KEY = "theme";
const DEFAULT_THEME: ThemePref = "system";

async function readTheme(): Promise<ThemePref> {
	const row = await db
		.select()
		.from(appSettings)
		.where(eq(appSettings.key, THEME_KEY))
		.get();
	const parsed = themeSchema.safeParse(row?.value);
	return parsed.success ? parsed.data : DEFAULT_THEME;
}

async function writeTheme(value: ThemePref): Promise<void> {
	await db
		.insert(appSettings)
		.values({ key: THEME_KEY, value })
		.onConflictDoUpdate({ target: appSettings.key, set: { value } })
		.run();
}

function broadcastTheme(value: ThemePref): void {
	for (const win of BrowserWindow.getAllWindows()) {
		win.webContents.send("settings:theme-changed", value);
	}
}

export const settingsRouter = router({
	getTheme: publicProcedure.query(async () => readTheme()),
	setTheme: publicProcedure
		.input(themeSchema)
		.mutation(async ({ input }) => {
			await writeTheme(input);
			broadcastTheme(input);
			return input;
		}),
});
```

Create `apps/desktop/src/renderer/lib/theme.ts`:

```ts
export type ThemePref = "system" | "light" | "dark";
export type ResolvedTheme = "light" | "dark";

const STORAGE_KEY = "theme-paint-cache";

export function resolveTheme(pref: ThemePref, systemDark: boolean): ResolvedTheme {
	if (pref === "system") return systemDark ? "dark" : "light";
	return pref;
}

export function applyTheme(resolved: ResolvedTheme): void {
	document.documentElement.dataset.theme = resolved;
	try {
		localStorage.setItem(STORAGE_KEY, resolved);
	} catch {
		// localStorage may be unavailable; not fatal
	}
}

export function readPaintCache(): ResolvedTheme | null {
	try {
		const v = localStorage.getItem(STORAGE_KEY);
		return v === "light" || v === "dark" ? v : null;
	} catch {
		return null;
	}
}

export function systemPrefersDark(): boolean {
	return matchMedia("(prefers-color-scheme: dark)").matches;
}

export function watchSystemTheme(cb: (dark: boolean) => void): () => void {
	const mql = matchMedia("(prefers-color-scheme: dark)");
	const handler = (e: MediaQueryListEvent) => cb(e.matches);
	mql.addEventListener("change", handler);
	return () => mql.removeEventListener("change", handler);
}
```

```ts
import { create } from "zustand";
import { trpc } from "../trpc";
import {
	type ResolvedTheme,
	type ThemePref,
	applyTheme,
	resolveTheme,
	systemPrefersDark,
	watchSystemTheme,
} from "../lib/theme";

interface ThemeState {
	pref: ThemePref;
	resolved: ResolvedTheme;
	setPref: (p: ThemePref) => Promise<void>;
	hydrate: () => Promise<void>;
}

export const useThemeStore = create<ThemeState>((set, get) => ({
	pref: "system",
	resolved: systemPrefersDark() ? "dark" : "light",

	setPref: async (p) => {
		await trpc.settings.setTheme.mutate(p);
		const resolved = resolveTheme(p, systemPrefersDark());
		applyTheme(resolved);
		set({ pref: p, resolved });
	},

	hydrate: async () => {
		const pref = await trpc.settings.getTheme.query();
		const resolved = resolveTheme(pref, systemPrefersDark());
		applyTheme(resolved);
		set({ pref, resolved });
	},
}));

// OS theme follow when pref === "system"
watchSystemTheme((dark) => {
	const { pref } = useThemeStore.getState();
	if (pref !== "system") return;
	const resolved: ResolvedTheme = dark ? "dark" : "light";
	applyTheme(resolved);
	useThemeStore.setState({ resolved });
});

// Sync across windows via main-process broadcast
window.electron?.settings?.onThemeChanged((value) => {
	const resolved = resolveTheme(value, systemPrefersDark());
	applyTheme(resolved);
	useThemeStore.setState({ pref: value, resolved });
});
```

```ts
import * as monaco from "monaco-editor";

export const EDITOR_THEME_DARK = "superiorswarm-dark";
export const EDITOR_THEME_LIGHT = "superiorswarm-light";

let registered = false;

export function ensureThemeRegistered(): void {
	if (registered) return;

	monaco.editor.defineTheme(EDITOR_THEME_DARK, {
		base: "vs-dark",
		inherit: true,
		rules: [],
		colors: {
			"editor.background": "#161618",
			"diffEditor.insertedTextBackground": "#1a3a2a80",
			"diffEditor.removedTextBackground": "#3a1a1a80",
			"diffEditor.insertedLineBackground": "#1a3a2a40",
			"diffEditor.removedLineBackground": "#3a1a1a40",
		},
	});

	monaco.editor.defineTheme(EDITOR_THEME_LIGHT, {
		base: "vs",
		inherit: true,
		rules: [],
		colors: {
			"editor.background": "#ffffff",
			"diffEditor.insertedTextBackground": "#aaf0c080",
			"diffEditor.removedTextBackground": "#ffc4c480",
			"diffEditor.insertedLineBackground": "#dcffe4",
			"diffEditor.removedLineBackground": "#ffe7e7",
		},
	});

	registered = true;
	syncMonacoTheme();
	startThemeWatcher();
}

function syncMonacoTheme(): void {
	const t = document.documentElement.dataset.theme === "light"
		? EDITOR_THEME_LIGHT
		: EDITOR_THEME_DARK;
	monaco.editor.setTheme(t);
}

let watcherStarted = false;
function startThemeWatcher(): void {
	if (watcherStarted) return;
	watcherStarted = true;
	const obs = new MutationObserver(syncMonacoTheme);
	obs.observe(document.documentElement, {
		attributes: true,
		attributeFilter: ["data-theme"],
	});
}

// Backwards-compat re-export for existing imports
export const EDITOR_THEME = EDITOR_THEME_DARK;
```
