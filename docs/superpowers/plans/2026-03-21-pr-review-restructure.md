- [ ] **Step 1: Create the shared cleanup function**

Create `apps/desktop/src/main/ai-review/cleanup.ts`:

```typescript
import { eq, and } from "drizzle-orm";
import { getDb } from "../db";
import { workspaces, worktrees, projects } from "../db/schema";
import { reviewDrafts } from "../db/schema-ai-review";
import { removeWorktree } from "../git/operations";

/**
 * Full cleanup of a review workspace: kills terminal, removes worktree,
 * deletes workspace record, dismisses all related drafts.
 * Used by: dismissReview, PR close detection, commit-poller on merge.
 */
export async function cleanupReviewWorkspace(workspaceId: string): Promise<void> {
	const db = getDb();

	// 1. Find workspace + worktree + project (need project.repoPath for removeWorktree)
	const workspace = db.select().from(workspaces).where(eq(workspaces.id, workspaceId)).get();
	if (!workspace) return;
	if (workspace.type !== "review") {
		throw new Error(`Cannot cleanup non-review workspace: ${workspaceId}`);
	}

	const project = db.select().from(projects).where(eq(projects.id, workspace.projectId)).get();

	// 2. Remove worktree from disk if it exists
	if (workspace.worktreeId && project) {
		const worktree = db.select().from(worktrees).where(eq(worktrees.id, workspace.worktreeId)).get();
		if (worktree?.path) {
			try {
				// removeWorktree(repoPath, worktreePath) — needs both args
				await removeWorktree(project.repoPath, worktree.path);
			} catch {
				// Worktree may already be gone — that's fine
			}
		}
		// Delete worktree DB record
		db.delete(worktrees).where(eq(worktrees.id, workspace.worktreeId)).run();
	}

	// 3. Dismiss all related drafts for this PR
	if (workspace.prProvider && workspace.prIdentifier) {
		db.update(reviewDrafts)
			.set({ status: "dismissed" })
			.where(eq(reviewDrafts.prIdentifier, workspace.prIdentifier))
			.run();
	}

	// 4. Delete workspace record
	db.delete(workspaces).where(eq(workspaces.id, workspaceId)).run();
}

/**
 * Find workspace ID for a given PR identifier. Used by pollers.
 */
export function findReviewWorkspaceByPR(
	projectId: string,
	prProvider: string,
	prIdentifier: string,
): string | undefined {
	const db = getDb();
	const ws = db
		.select({ id: workspaces.id })
		.from(workspaces)
		.where(and(
			eq(workspaces.projectId, projectId),
			eq(workspaces.prProvider, prProvider),
			eq(workspaces.prIdentifier, prIdentifier),
		))
		.get();
	return ws?.id;
}
```

getOrCreateReview: publicProcedure
	.input(z.object({
		projectId: z.string(),
		prProvider: z.enum(["github", "bitbucket"]),
		prIdentifier: z.string(),
		prTitle: z.string(),
		sourceBranch: z.string(),
		targetBranch: z.string(),
	}))
	.mutation(async ({ input }) => {
		const db = getDb();

		// 1. Check for existing review workspace
		let workspace = db.select().from(workspaces)
			.where(and(
				eq(workspaces.projectId, input.projectId),
				eq(workspaces.prProvider, input.prProvider),
				eq(workspaces.prIdentifier, input.prIdentifier),
			)).get();

		// 2. Create if not exists
		if (!workspace) {
			const id = nanoid();
			const now = new Date();
			const name = `PR #${input.prIdentifier.split("#")[1]}: ${input.prTitle}`;
			db.insert(workspaces).values({
				id,
				projectId: input.projectId,
				name,
				type: "review",
				prProvider: input.prProvider,
				prIdentifier: input.prIdentifier,
				createdAt: now,
				updatedAt: now,
			}).run();
			workspace = db.select().from(workspaces).where(eq(workspaces.id, id)).get()!;
		}

		// 3. Ensure worktree exists
		// (absorb logic from review-workspaces.ts:createWorktree lines 167-249)
		const project = db.select().from(projects).where(eq(projects.id, input.projectId)).get();
		if (!project) throw new Error("Project not found");

		if (!workspace.worktreeId) {
			// Compute worktree path using worktreeBasePath (same pattern as existing code)
			const sanitizedId = input.prIdentifier.replace(/[^a-zA-Z0-9-]/g, "-");
			const wtPath = join(worktreeBasePath(project.repoPath), `pr-review-${sanitizedId}`);

			// Clean up stale worktree at same path if it exists
			const { existsSync, rmSync } = await import("node:fs");
			if (existsSync(wtPath)) {
				try {
					await removeWorktree(project.repoPath, wtPath);
				} catch {
					rmSync(wtPath, { recursive: true, force: true });
					const { default: simpleGit } = await import("simple-git");
					await simpleGit(project.repoPath).raw(["worktree", "prune"]).catch(() => {});
				}
			}

			// checkoutBranchWorktree(repoPath, worktreePath, branch) — returns void
			await checkoutBranchWorktree(project.repoPath, wtPath, input.sourceBranch);

			const now = new Date();
			const worktreeId = nanoid();
			db.insert(worktrees).values({
				id: worktreeId,
				projectId: input.projectId,
				path: wtPath,
				branch: input.sourceBranch,
				baseBranch: input.targetBranch,
				createdAt: now,
				updatedAt: now,
			}).run();

			db.update(workspaces)
				.set({ worktreeId, updatedAt: now })
				.where(eq(workspaces.id, workspace.id))
				.run();

			workspace = db.select().from(workspaces).where(eq(workspaces.id, workspace.id)).get()!;
		} else {
			// Worktree exists — update to latest
			const worktree = db.select().from(worktrees).where(eq(worktrees.id, workspace.worktreeId)).get();
			if (worktree?.path) {
				await execGit(worktree.path, ["fetch", "origin"]);
				await execGit(worktree.path, ["reset", "--hard", `origin/${input.sourceBranch}`]);
			}
		}

		// 4. Return workspace with worktree path
		const worktree = workspace.worktreeId
			? db.select().from(worktrees).where(eq(worktrees.id, workspace.worktreeId)).get()
			: null;

		return { ...workspace, worktreePath: worktree?.path ?? null };
	}),

export function getCachedPRs(projectId?: string): CachedPR[] {
	const all = Array.from(cache.values());
	return projectId ? all.filter((pr) => pr.projectId === projectId) : all;
}

export function onNewPRDetected(handler: (pr: CachedPR) => void): void {
	onNewPR = handler;

export function onPRClosedDetected(handler: (pr: CachedPR) => void): void {

async function pollAllProjects(): Promise<void> {
	const db = getDb();
	const allProjects = db.select().from(projects).all();

	// 1. Fetch GitHub PRs (uses getMyPRs from src/main/github/github.ts:306)
	//    This returns GitHubPR[] — map to CachedPR format
	try {
		const ghPRs = await getMyPRs();
		for (const pr of ghPRs) {
			// Match PR to project by githubOwner + githubRepo
			const project = allProjects.find(
				(p) => p.githubOwner === pr.repoOwner && p.githubRepo === pr.repoName
			);
			if (!project) continue;

			const identifier = `${pr.repoOwner}/${pr.repoName}#${pr.number}`;
			const existing = cache.get(identifier);
			const entry: CachedPR = {
				provider: "github",
				identifier,
				number: pr.number,
				title: pr.title,
				state: pr.state as CachedPR["state"],
				sourceBranch: pr.branchName,
				targetBranch: project.defaultBranch ?? "main",
				author: { login: pr.author, avatarUrl: pr.authorAvatarUrl ?? "" },
				reviewers: [], // Enriched separately if needed
				ciStatus: null,
				commentCount: pr.commentCount ?? 0,
				changedFiles: pr.changedFiles ?? 0,
				additions: pr.additions ?? 0,
				deletions: pr.deletions ?? 0,
				updatedAt: pr.updatedAt ?? new Date().toISOString(),
				repoOwner: pr.repoOwner,
				repoName: pr.repoName,
				projectId: project.id,
			};

			cache.set(identifier, entry);

			// Detect new PRs (not previously in cache)
			if (!existing && onNewPR) onNewPR(entry);
			// Detect closed/merged PRs
			if (existing?.state === "open" && (entry.state === "merged" || entry.state === "closed")) {
				if (onPRClosed) onPRClosed(entry);
			}
		}
	} catch {
		// GitHub auth may not be configured — that's fine
	}

	// 2. Fetch Bitbucket PRs (uses getMyPullRequests and getReviewRequests
	//    from src/main/atlassian/bitbucket.ts:66,93)
	try {
		const [bbOwned, bbReview] = await Promise.all([getMyPullRequests(), getReviewRequests()]);
		// Deduplicate (PR may appear in both owned and review lists)
		const seen = new Set<string>();
		for (const pr of [...bbOwned, ...bbReview]) {
			const identifier = `${pr.destination.repository.full_name}#${pr.id}`;
			if (seen.has(identifier)) continue;
			seen.add(identifier);

			// Match to project by Bitbucket workspace/repo slug
			const [wsSlug, repoSlug] = pr.destination.repository.full_name.split("/");
			const project = allProjects.find(
				(p) => p.bitbucketWorkspace === wsSlug && p.bitbucketRepo === repoSlug
			);
			if (!project) continue;

			const existing = cache.get(identifier);
			const entry: CachedPR = {
				provider: "bitbucket",
				identifier,
				number: pr.id,
				title: pr.title,
				state: pr.state as CachedPR["state"], // "OPEN" | "MERGED" | "DECLINED"
				sourceBranch: pr.source.branch.name,
				targetBranch: pr.destination.branch.name,
				author: {
					login: pr.author.display_name,
					avatarUrl: pr.author.links?.avatar?.href ?? "",
				},
				reviewers: (pr.reviewers ?? []).map((r) => ({
					login: r.display_name,
					avatarUrl: r.links?.avatar?.href ?? "",
					state: r.approved ? "APPROVED" : "PENDING",
				})),
				ciStatus: null,
				commentCount: pr.comment_count ?? 0,
				changedFiles: 0, // Bitbucket doesn't return this in list endpoint
				additions: 0,
				deletions: 0,
				updatedAt: pr.updated_on ?? new Date().toISOString(),
				repoOwner: wsSlug,
				repoName: repoSlug,
				projectId: project.id,
			};

			cache.set(identifier, entry);
			if (!existing && onNewPR) onNewPR(entry);
			if (existing?.state === "open" && (entry.state === "merged" || entry.state === "declined")) {
				if (onPRClosed) onPRClosed(entry);
			}
		}
	} catch {
		// Bitbucket auth may not be configured
	}

	// 3. Clean up cache entries for PRs no longer returned by APIs
	// (they may have been closed/merged between polls)
}

export function startPolling(intervalMs = 60_000): void {
	if (pollInterval) return;
	pollAllProjects(); // Initial poll
	pollInterval = setInterval(pollAllProjects, intervalMs);
}

export function stopPolling(): void {
	if (pollInterval) {
		clearInterval(pollInterval);
		pollInterval = null;
	}
}

export async function refreshNow(): Promise<void> {
	await pollAllProjects();
}
